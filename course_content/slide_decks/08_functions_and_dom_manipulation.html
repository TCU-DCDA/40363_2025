Let's Write Functions

// Simple function - no input needed
function sayHello() {
  console.log("Hello, world!");
}
 
// Function with input (parameter)
function calculateAge(birthYear) {
  let currentYear = 2025;
  return currentYear - birthYear;
}
 
// Using our functions
sayHello();
// Prints: Hello, world!
 
let age = calculateAge(2000);
// Returns: 25

The DOM: JavaScript's Map to HTML

DOM = Document Object Model

Your HTML:

<h1 id="title">My Website</h1>
<p id="message">Welcome!</p>
<button id="clickMe">Click Me</button>

JavaScript can find and change these elements!

// Find an element by its ID
 
let titleElement = document.getElementById("title");

Finding HTML Elements

// Find element by ID (most common)
let title = document.getElementById("title");
 
// Find element by class name
let buttons = document.getElementsByClassName("btn");
 
// Find element by tag name
let paragraphs = document.getElementsByTagName("p");

Pro Tip:

Always use IDs for elements you want to change with JavaScript

IDs must be unique on each page!

Changing What's on the Page

// Change the text inside an element
let message = document.getElementById("message");
message.textContent = "Hello from JavaScript!";
 
// Change the HTML inside an element
let title = document.getElementById("title");
title.innerHTML = "<strong>Updated Title</strong>";
 
// Change CSS styles
title.style.color = "blue";
title.style.fontSize = "2rem";

This text will change!

Making Things Interactive

Events happen when users interact with your page

  • click - user clicks something
  • keypress - user types
  • mouseover - user hovers
  • load - page finishes loading
// Find the button
let button = document.getElementById("clickMe");
 
// Listen for click events
button.addEventListener("click", function() {
  alert("Button was clicked!");
});

Putting It All Together

// HTML:
// <button id="colorButton">Change Color</button>
// <p id="colorText">This text will change color</p>
 
function changeColor() {
  let textElement = document.getElementById("colorText");
  let colors = ["red", "blue", "green", "purple", "orange"];
  let randomColor = colors[Math.floor(Math.random() * colors.length)];
 
  textElement.style.color = randomColor;
  textElement.textContent = "Now I'm " + randomColor + "!";
}
 
// Connect the function to the button
let button = document.getElementById("colorButton");
button.addEventListener("click", changeColor);

This text will change color!
