Arrays: Your Digital Lists

Arrays are like numbered containers that hold multiple pieces of information

0
"Pizza"

let favoriteFoods = ["Pizza", "Burgers", "Tacos", "Pasta"];
 
// Access items by their position (starting from 0)
console.log(favoriteFoods[0]);
// Prints: "Pizza"

Working with Arrays

let movies = ["Avengers", "Star Wars"];
 
// Add to the end
movies.push("Spider-Man");
 
// Remove from the end
let lastMovie = movies.pop();
 
// Check how many items
console.log(movies.length);
// Prints: 2

Objects: Labeled Information

Objects organize information using names instead of numbers

Student Information

name: "Sarah"
age: 20
major: "Computer Science"
enrolled: true
let student = {
  name: "Sarah",
  age: 20,
  major: "Computer Science",
  enrolled: true
};

Getting Information from Objects

let restaurant = {
  name: "Tony's Pizza",
  rating: 4.5,
  cuisine: "Italian",
  delivery: true
};
 
// Dot notation (most common)
console.log(restaurant.name);
// Prints: "Tony's Pizza"
 
// Bracket notation
console.log(restaurant["rating"]);
// Prints: 4.5

When to Use Arrays vs Objects

Arrays

Use when:

  • Order matters
  • Similar items
  • Need to count
  • Lists of things

Example: Shopping list, favorite movies

Objects

Use when:

  • Need labels
  • Different types of data
  • Describing something
  • Related information

Example: Person info, restaurant details

Combining Arrays and Objects

The power combination for managing real data!

let favoriteRestaurants = [
  {
    name: "Mario's Pizza",
    rating: 5,
    cuisine: "Italian"
  },
  {
    name: "Sushi Express",
    rating: 4,
    cuisine: "Japanese"
  }
];
 
// Access first restaurant's name
console.log(favoriteRestaurants[0].name);
// Prints: "Mario's Pizza"

Finding What You Need

let books = [
  { title: "Harry Potter", genre: "Fantasy", rating: 5 },
  { title: "The Martian", genre: "Sci-Fi", rating: 4 },
  { title: "Pride and Prejudice", genre: "Romance", rating: 4 }
];
 
// Find all books with 5-star rating
for (let i = 0; i < books.length; i++) {
  if (books[i].rating === 5) {
    console.log(books[i].title);
  }
}
// Prints: "Harry Potter"

Building Your Collection

let myFavorites = [];
 
// Add new favorite restaurant
function addFavorite(name, rating, cuisine) {
  let newRestaurant = {
    name: name,
    rating: rating,
    cuisine: cuisine,
    dateAdded: new Date()
  };
 
  myFavorites.push(newRestaurant);
}
 
// Use the function
addFavorite("Burger Palace", 4, "American");
