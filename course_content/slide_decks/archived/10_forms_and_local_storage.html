Processing Form Data with JavaScript

let form = document.getElementById("restaurantForm");
 
form.addEventListener("submit", function(event) {
  event.preventDefault();
  // Prevent page refresh
 
  let name = document.getElementById("name").value;
  let rating = document.getElementById("rating").value;
 
  console.log("Name: " + name);
  console.log("Rating: " + rating);
});

Validating User Input

function validateForm(name, rating) {
  if (name.trim() === "") {
    alert("Please enter a restaurant name");
    return false;
  }
 
  if (rating < 1 || rating > 5) {
    alert("Rating must be between 1 and 5");
    return false;
  }
 
  return true;
}

Good validation: Helps users enter correct information and prevents errors in your app!

Saving and Loading Data

// Save a single value
localStorage.setItem("userName", "Sarah");
 
// Get a value back
let name = localStorage.getItem("userName");
console.log(name);
// Prints: "Sarah"
 
// Save an array (convert to string first)
let favorites = ["Pizza", "Burgers", "Tacos"];
localStorage.setItem("favorites", JSON.stringify(favorites));
 
// Get array back (convert from string)
let saved = JSON.parse(localStorage.getItem("favorites"));

Building a Persistent Favorites List

function addRestaurant(name, rating, cuisine) {
  let favorites = loadFavorites();
 
  let newRestaurant = {
    name: name,
    rating: rating,
    cuisine: cuisine,
    id: Date.now()
  };
 
  favorites.push(newRestaurant);
  saveFavorites(favorites);
}
 
function saveFavorites(favorites) {
  localStorage.setItem("restaurants", JSON.stringify(favorites));
}

Loading Data When Page Opens

function loadFavorites() {
  let saved = localStorage.getItem("restaurants");
 
  if (saved) {
    return JSON.parse(saved);
  } else {
    return [];
    // Return empty array if no data
  }
}
 
// Load data when page first opens
window.addEventListener("load", function() {
  let favorites = loadFavorites();
  displayFavorites(favorites);
});
